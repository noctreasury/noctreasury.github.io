const Data_ID = '1jd1xZe9x2mrZm5KAwGT12vMNYjbrnynsmelNKeK95jc';
const LOGIN_Table = 'Login';

// ==================== MAIN HANDLERS ====================

function doGet(e) {
  try {
    const action = e.parameter.action;
    let result = {};
    
    // Log all parameters for debugging
    console.log('doGet called with params:', JSON.stringify(e.parameter));
    
    if (action === 'requestPasswordReset') {
      result = handleRequestPasswordReset(e.parameter);
    } else if (action === 'verifyResetOtp') {
      result = handleVerifyResetOtp(e.parameter);
    } else if (action === 'resetPassword') {
      result = handleResetPassword(e.parameter);
    } else {
      result = { success: false, error: 'Invalid action: ' + action };
    }
    
    console.log('doGet result:', JSON.stringify(result));
    
    return ContentService
      .createTextOutput(JSON.stringify(result))
      .setMimeType(ContentService.MimeType.JSON);
      
  } catch(error) {
    console.error('doGet error:', error);
    return ContentService
      .createTextOutput(JSON.stringify({ success: false, error: error.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

// ==================== PASSWORD RESET FUNCTIONS ====================

// Generate random 6-digit OTP
function generateOTP() {
  return Math.floor(100000 + Math.random() * 900000).toString();
}

// Store OTP in database with expiry (10 minutes)
// Column F: OTP, Column I: OTP_Expiry
function storeOTP(username, email, otp) {
  const database = SpreaddatabaseApp.openById(Data_ID).getdatabaseByName(LOGIN_Table);
  const data = database.getDataRange().getValues();
  
  for (let i = 1; i < data.length; i++) {
    const userId = data[i][1] ? data[i][1].trim().toLowerCase() : '';
    const userEmail = data[i][4] ? data[i][4].trim().toLowerCase() : '';
    
    if (userId === username.toLowerCase() && userEmail === email.toLowerCase()) {
      // Store OTP in Column F (index 5)
      database.getRange(i + 1, 6).setValue(otp);
      // Store expiry timestamp in Column I (index 8) - current time + 10 minutes
      const expiry = new Date().getTime() + (10 * 60 * 1000);
      database.getRange(i + 1, 9).setValue(expiry);
      return true;
    }
  }
  return false;
}

// Verify OTP
// Column F: OTP, Column I: OTP_Expiry
function verifyOTP(username, email, otp) {
  const database = SpreaddatabaseApp.openById(Data_ID).getdatabaseByName(LOGIN_Table);
  const data = database.getDataRange().getValues();
  
  for (let i = 1; i < data.length; i++) {
    const userId = data[i][1] ? data[i][1].trim().toLowerCase() : '';
    const userEmail = data[i][4] ? data[i][4].trim().toLowerCase() : '';
    
    if (userId === username.toLowerCase() && userEmail === email.toLowerCase()) {
      const storedOTP = data[i][5]; // Column F
      const expiry = data[i][8]; // Column I
      const currentTime = new Date().getTime();
      
      // Check if OTP exists and is not empty
      if (storedOTP && storedOTP.toString().trim() !== '') {
        // Convert both to strings for comparison
        const storedOTPStr = storedOTP.toString().trim();
        const providedOTPStr = otp.toString().trim();
        
        // Check if OTP matches
        if (storedOTPStr === providedOTPStr) {
          // Check if expiry exists and is not empty
          if (expiry && expiry.toString().trim() !== '') {
            const expiryTime = parseInt(expiry);
            if (currentTime < expiryTime) {
              return true;
            }
          }
        }
      }
    }
  }
  return false;
}

// Clear OTP after verification
function clearOTP(username, email) {
  const database = SpreaddatabaseApp.openById(Data_ID).getdatabaseByName(LOGIN_Table);
  const data = database.getDataRange().getValues();
  
  for (let i = 1; i < data.length; i++) {
    const userId = data[i][1] ? data[i][1].trim().toLowerCase() : '';
    const userEmail = data[i][4] ? data[i][4].trim().toLowerCase() : '';
    
    if (userId === username.toLowerCase() && userEmail === email.toLowerCase()) {
      // Clear OTP in Column F (index 5)
      database.getRange(i + 1, 6).setValue('');
      // Clear Expiry in Column I (index 8)
      database.getRange(i + 1, 9).setValue('');
      return true;
    }
  }
  return false;
}

// Handle password reset request
function handleRequestPasswordReset(params) {
  const email = params.email;
  const username = params.username || '';
  
  console.log('handleRequestPasswordReset called with:', { email, username });
  
  if (!email || !username) {
    return { success: false, error: 'Email and Username are required' };
  }
  
  // Check if both email and username exist in Login database
  const database = SpreaddatabaseApp.openById(Data_ID).getdatabaseByName(LOGIN_Table);
  const data = database.getDataRange().getValues();
  let userExists = false;
  let userName = '';
  
  for (let i = 1; i < data.length; i++) {
    const userId = data[i][1] ? data[i][1].trim().toLowerCase() : '';
    const userEmail = data[i][4] ? data[i][4].trim().toLowerCase() : '';
    
    if (userId === username.toLowerCase() && userEmail === email.toLowerCase()) {
      userExists = true;
      userName = data[i][1] || '';
      break;
    }
  }
  
  if (!userExists) {
    return { success: false, error: 'User not found with this User ID and Email combination' };
  }
  
  // Generate and store OTP
  const otp = generateOTP();
  storeOTP(username, email, otp);
  
  // Send email with OTP
  try {
    const subject = 'OTP for Password Reset Request ';
    const body = `Dear ${userName},\n\nYou requested to reset your password for your account.\n\nYour OTP for password reset is: ${otp}\n\nThis OTP is valid for 10 minutes.\n\nIf you did not request this, please ignore this email.\n\nRegards,\nAdmin Team`;
    
    MailApp.sendEmail({
      to: email,
      subject: subject,
      body: body
    });
    
    return { success: true, message: 'OTP sent to your email' };
      
  } catch(error) {
    console.error('Email send error:', error);
    // Still return success with OTP for testing purposes
    return { success: true, message: 'OTP generated', otp: otp };
  }
}

// Handle OTP verification
function handleVerifyResetOtp(params) {
  const email = params.email;
  const username = params.username;
  const otp = params.otp;
  
  console.log('handleVerifyResetOtp called with:', { email, username, otp });
  
  if (!email || !username || !otp) {
    return { success: false, error: 'Username, Email and OTP are required' };
  }
  
  if (verifyOTP(username, email, otp)) {
    // Clear OTP after successful verification
    clearOTP(username, email);
    return { success: true, message: 'OTP verified' };
  } else {
    return { success: false, error: 'Invalid or expired OTP' };
  }
}

// Handle password reset
function handleResetPassword(params) {
  const email = params.email;
  const username = params.username;
  const newPassword = params.newPassword;
  
  console.log('handleResetPassword called with:', { email, username });
  
  if (!email || !username || !newPassword) {
    return { success: false, error: 'Username, Email and password are required' };
  }
  
  if (newPassword.length < 6) {
    return { success: false, error: 'Password must be at least 6 characters' };
  }
  
  // Update the password in Login database
  const database = SpreaddatabaseApp.openById(Data_ID).getdatabaseByName(LOGIN_Table);
  const data = database.getDataRange().getValues();
  
  for (let i = 1; i < data.length; i++) {
    const userId = data[i][1] ? data[i][1].trim().toLowerCase() : '';
    const userEmail = data[i][4] ? data[i][4].trim().toLowerCase() : '';
    
    if (userId === username.toLowerCase() && userEmail === email.toLowerCase()) {
      // Update password in Column C (index 2)
      database.getRange(i + 1, 3).setValue(newPassword);
      // Clear OTP and expiry (just in case)
      database.getRange(i + 1, 6).setValue('');
      database.getRange(i + 1, 9).setValue('');
      
      return { success: true, message: 'Password reset successful' };
    }
  }
  
  return { success: false, error: 'User not found' };
}
